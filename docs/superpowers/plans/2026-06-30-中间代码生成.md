完整内容写入 `C/src/irvm/irvm.c`。设计：第一遍扫描建标号表（label 名→行号）；变量环境用 pjw 哈希（变量名→long 值）；PC 程序计数器驱动；CALL 用调用栈保存返回行号。

```c
/*
 * irvm.c - 中间代码虚拟机（实践 3 验证工具）
 * 读 .ir 文本文件 + stdin（每个 READ 一行整数），执行 PDF 表 4.6 全部 17 种语句，
 * 把所有 WRITE 的值逐行打到 stdout。仅用于自动验证，不是交付物。
 *
 * 语义约定：
 *   - 变量/临时变量统一进哈希表，按 name 取值（int long）。
 *   - DEC v size：给 v 分配 size 字节的"内存"（用哈希项的 mem 指针模拟，&v 取其地址）。
 *   - 数组/结构体：&v 取首地址，*p 取值，*p := x 写值。地址用全局 mem[] 数组的偏移表示。
 *   - 整数运算用 long，按 C 语义（截断除法、模用 a-(a/b)*b 算）。
 */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_INSTR 200000
#define MEM_SIZE  (1<<20)
#define NHASH     0x3fff

typedef struct { char *op; long val; void *mem; } Var;  /* val=标量值；mem=DEC分配的地址 */
typedef struct { char *text; } Line;

static Line  prog[MAX_INSTR];
static int   nprog = 0;
static long  mem[MEM_SIZE];          /* 模拟内存，存数组/结构体元素 */
static int   mem_top = 1;            /* 下一个可用地址（0 留作 NULL） */

/* 标号表：label 名 → prog 下标 */
static char  *lname[NHASH];
static int    lline[NHASH];
static int    nlabel = 0;

static Var   *vtab[NHASH];           /* 变量哈希表 */

static unsigned pjw_hash(const char *s) {
    unsigned h = 0, g;
    for (; *s; ++s) {
        h = (h << 4) + *s;
        if ((g = h & 0xf0000000)) { h ^= g >> 24; h &= ~g; }
    }
    return h % NHASH;
}

static Var *var_get(const char *name) {
    unsigned h = pjw_hash(name);
    Var *v = vtab[h];
    while (v) {
        if (v->op && strcmp(v->op, name) == 0) return v;
        v = (Var *)((void **)v + 1);  /* 简化：实际用链表，见 Step 2 */
    }
    return NULL;
}
```

> ⚠️ Step 1 的 `var_get` 链表遍历是占位写法。完整实现用冲突链表：`vtab[h]` 指向链表头节点，节点含 `next` 指针。下面 Step 2 给出完整可编译版本。

- [ ] **Step 2: 写完整可编译的 irvm.c**

把 Step 1 的占位换成完整实现。完整文件结构（直接写入 `C/src/irvm/irvm.c`）：

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_INSTR 200000
#define MEM_SIZE  (1<<20)
#define NHASH     0x3fff

typedef struct Var {
    char *name;
    long  val;          /* 标量值 */
    int   mem_addr;     /* DEC 分配的内存地址（0=未分配） */
    struct Var *next;
} Var;

typedef struct { char *text; } Line;

static Line  prog[MAX_INSTR];
static int   nprog = 0;
static long  mem[MEM_SIZE];
static int   mem_top = 1;
static Var  *vtab[NHASH];

/* 标号表 */
static char *lname[NHASH];
static int   lline[NHASH];
static int   nlabel = 0;

static unsigned pjw_hash(const char *s) {
    unsigned h = 0, g;
    for (; *s; ++s) {
        h = (h << 4) + *s;
        if ((g = h & 0xf0000000)) { h ^= g >> 24; h &= ~g; }
    }
    return h % NHASH;
}

static Var *var_lookup(const char *name) {
    unsigned h = pjw_hash(name);
    Var *v = vtab[h];
    while (v) { if (strcmp(v->name, name) == 0) return v; v = v->next; }
    return NULL;
}

static Var *var_intern(const char *name) {
    Var *v = var_lookup(name);
    if (v) return v;
    unsigned h = pjw_hash(name);
    v = (Var *)calloc(1, sizeof(Var));
    v->name = strdup(name);
    v->next = vtab[h];
    vtab[h] = v;
    return v;
}

/* 解析操作数：#const → 常量值；&name → 变量地址；name → 变量值；*name → 地址处值 */
static long eval_operand(const char *tok) {
    if (tok[0] == '#') return atol(tok + 1);
    if (tok[0] == '&') { Var *v = var_intern(tok + 1); return v->mem_addr; }
    if (tok[0] == '*') { Var *v = var_intern(tok + 1); return mem[v->val]; }
    Var *v = var_intern(tok);
    return v->val;
}

/* 写目标：tok 是左值。普通变量/临时 → 写 val；*p → 写 mem[p->val] */
static void store_operand(const char *tok, long value) {
    if (tok[0] == '*') {
        Var *v = var_intern(tok + 1);
        mem[v->val] = value;
        return;
    }
    Var *v = var_intern(tok);
    v->val = value;
}

static int find_label(const char *name) {
    int i;
    for (i = 0; i < nlabel; ++i)
        if (strcmp(lname[i], name) == 0) return lline[i];
    return -1;
}

/* 把一行按空格切成 token（最多 8 个）。返回 token 数。
   注意：原文 "x := y + z" 切出 ["x",":=","y","+","z"]。 */
static int tokenize(char *line, char *tok[], int max) {
    int n = 0;
    char *p = strtok(line, " \t\r\n");
    while (p && n < max) { tok[n++] = p; p = strtok(NULL, " \t\r\n"); }
    return n;
}

/* 跳过 := 运算符所在位置。如 tok=[x,:=,y,+,z]，op_idx 返回 2（y 的位置）。 */
/* 由调用方按语句类型直接定位，这里不集中处理。 */

int main(int argc, char **argv) {
    if (argc < 2) { fprintf(stderr, "usage: irvm <file.ir>\n"); return 1; }
    FILE *f = fopen(argv[1], "r");
    if (!f) { perror("fopen"); return 1; }

    char buf[1024];
    while (fgets(buf, sizeof(buf), f)) {
        /* 去行尾换行，存原文 */
        prog[nprog].text = strdup(buf);
        nprog++;
    }
    fclose(f);

    /* 第一遍：建标号表 */
    int i;
    for (i = 0; i < nprog; ++i) {
        char tmp[1024]; strcpy(tmp, prog[i].text);
        char *tok[8];
        int n = tokenize(tmp, tok, 8);
        if (n >= 2 && strcmp(tok[0], "LABEL") == 0) {
            /* "LABEL name :" → tok[1]=name，tok[2]=":"（被切掉） */
            lname[nlabel] = strdup(tok[1]);
            lline[nlabel] = i;
            nlabel++;
        }
    }

    /* 第二遍：执行 */
    int pc = 0;
    while (pc < nprog) {
        char tmp[1024]; strcpy(tmp, prog[pc].text);
        char *tok[8];
        int n = tokenize(tmp, tok, 8);
        if (n == 0) { pc++; continue; }
        char *kw = tok[0];

        if (strcmp(kw, "LABEL") == 0 || strcmp(kw, "FUNCTION") == 0) {
            pc++; continue;
        }
        if (strcmp(kw, "GOTO") == 0) {
            pc = find_label(tok[1]); continue;
        }
        if (strcmp(kw, "IF") == 0) {
            /* IF x relop y GOTO z → tok=[IF,x,relop,y,GOTO,z] */
            long a = eval_operand(tok[1]);
            long b = eval_operand(tok[3]);
            char *r = tok[2]; int cond = 0;
            if (strcmp(r,"<")==0) cond = a < b;
            else if (strcmp(r,"<=")==0) cond = a <= b;
            else if (strcmp(r,">")==0) cond = a > b;
            else if (strcmp(r,">=")==0) cond = a >= b;
            else if (strcmp(r,"==")==0) cond = a == b;
            else if (strcmp(r,"!=")==0) cond = a != b;
            if (cond) pc = find_label(tok[5]); else pc++;
            continue;
        }
        if (strcmp(kw, "RETURN") == 0) {
            /* 顶层 RETURN 退出；函数 RETURN 见 CALL 栈处理 */
            return (int)eval_operand(tok[1]);
        }
        if (strcmp(kw, "DEC") == 0) {
            /* DEC v size → 给 v 分配 size/4 个 long 槽 */
            Var *v = var_intern(tok[1]);
            int size = atoi(tok[2]);
            v->mem_addr = mem_top;
            mem_top += (size + sizeof(long) - 1) / sizeof(long);
            pc++; continue;
        }
        if (strcmp(kw, "ARG") == 0) {
            /* 压入参数栈，CALL 时弹给 PARAM */
            /* 简化：维护全局 arg 栈，PARAM 逆序取。这里只存值。 */
            /* 实际见 Step 3 的 arg_stack。 */
            pc++; continue;
        }
        if (strcmp(kw, "READ") == 0) {
            long x; if (scanf("%ld", &x) != 1) x = 0;
            store_operand(tok[1], x); pc++; continue;
        }
        if (strcmp(kw, "WRITE") == 0) {
            printf("%ld\n", eval_operand(tok[1])); pc++; continue;
        }
        if (strcmp(kw, "PARAM") == 0) { pc++; continue; }
        if (strcmp(kw, "CALL") == 0) {
            /* x := CALL f 或 CALL f。read/write 是 builtin 已特判翻译，这里不处理。 */
            pc++; continue;
        }
        /* 赋值/运算：tok[1] 应为 ":=" */
        if (n >= 2 && strcmp(tok[1], ":=") == 0) {
            char *dst = tok[0];
            if (n == 3) {
                /* x := y */
                store_operand(dst, eval_operand(tok[2]));
            } else if (n == 5) {
                /* x := y op z */
                long a = eval_operand(tok[2]);
                long b = eval_operand(tok[4]);
                char *op = tok[3]; long r = 0;
                if (strcmp(op,"+")==0) r = a + b;
                else if (strcmp(op,"-")==0) r = a - b;
                else if (strcmp(op,"*")==0) r = a * b;
                else if (strcmp(op,"/")==0) r = (b != 0) ? a / b : 0;
                store_operand(dst, r);
            }
            pc++; continue;
        }
        pc++;
    }
    return 0;
}
```

```c
/*
 * operand.h - IR 操作数（实践 3）
 * PDF 4.2.4 的 place 可以是变量/临时变量/立即数/标号/地址。统一为 Operand。
 */
#ifndef OPERAND_H
#define OPERAND_H

typedef enum {
    OP_VAR,       /* 源码变量：name = "v_a" */
    OP_TEMP,      /* 临时变量：name = "t1" */
    OP_CONST,     /* 立即数：value，打印为 "#value" */
    OP_LABEL,     /* 标号：name = "label1"（仅 IR_IF/GOTO/LABEL 用） */
    OP_ADDR       /* &name：变量地址（数组/结构体参数传引用） */
} OperandKind;

typedef struct {
    OperandKind kind;
    char *name;     /* OP_VAR/OP_TEMP/OP_LABEL/OP_ADDR */
    int   value;    /* OP_CONST */
} Operand;

typedef Operand *OperandPtr;

/* 生成器（全局计数器） */
Operand new_temp(void);                       /* 返回 OP_TEMP，name=t1,t2,... */
Operand new_label(void);                      /* 返回 OP_LABEL，name=label1,... */
Operand new_const(int value);                 /* 返回 OP_CONST */
Operand new_var(const char *src_name);        /* 返回 OP_VAR，name=v_src_name */
```

```c
#include "translate.h"
#include "ir.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define NHASH 0x3fff

static VarEntry *vtab[NHASH];

static unsigned pjw_hash(const char *s) {
    unsigned h = 0, g;
    for (; *s; ++s) {
        h = (h << 4) + *s;
        if ((g = h & 0xf0000000)) { h ^= g >> 24; h &= ~g; }
    }
    return h % NHASH;
}

VarEntry *translate_lookup_var(const char *src_name) {
    unsigned h = pjw_hash(src_name);
    VarEntry *e = vtab[h];
    while (e) { if (strcmp(e->src_name, src_name) == 0) return e; e = e->next; }
    return NULL;
}

void translate_insert_var(const char *src_name, Type type) {
    if (translate_lookup_var(src_name)) return;  /* 假设4不重名，已存在则跳过 */
    unsigned h = pjw_hash(src_name);
    VarEntry *e = (VarEntry *)calloc(1, sizeof(VarEntry));
    e->src_name = strdup(src_name);
    char buf[128]; snprintf(buf, sizeof(buf), "v_%s", src_name);
    e->ir_name = strdup(buf);
    e->type = type;
    e->next = vtab[h];
    vtab[h] = e;
}

        return new_basic(strcmp(child->sval, "float") == 0 ? 1 : 0);

    /* VarDec → ID  |  VarDec LB INT RB，递归到最内层 ID */
    while (strcmp(vardec->children[0]->name, "VarDec") == 0)
        vardec = vardec->children[0];
    return vardec->children[0];  /* ID 节点 */

Type tr_vardec_to_type(Node *vardec, Type base) {
    /* VarDec → ID ：返回 base
       VarDec → VarDec LB INT RB：递归构造数组，最外层=第一维。
       注意 PDF 数组类型约定：new_array(elem,size)，最外层维度在链表头。 */
    if (strcmp(vardec->children[0]->name, "VarDec") != 0)
        return base;
    Node *inner = vardec->children[0];

    /* 修正：VarDec → VarDec LB INT RB，children[0]=VarDec, [1]=LB, [2]=INT, [3]=RB */
    Type inner_type = tr_vardec_to_type(inner, base);

static FieldList collect_struct_fields(Node *structspec) {
    /* StructSpecifier → STRUCT OptTag LC DefList RC
       逐个 Def → Specifier DecList → Dec → VarDec，按出现顺序累加域。 */
    Node *deflist = NULL;
    int i;
    for (i = 0; i < structspec->nchild; ++i) {
        if (structspec->children[i] && strcmp(structspec->children[i]->name, "DefList") == 0) {
            deflist = structspec->children[i]; break;
        }
    }
    FieldList head = NULL, tail = NULL;
    while (deflist && deflist->children[0]) {
        Node *def = deflist->children[0];          /* Def → Specifier DecList SEMI */
        Node *spec = def->children[0];
        Node *declist = def->children[1];
        Type base = tr_specifier_to_type(spec);
        while (declist) {
            /* DecList → Dec  |  Dec COMMA DecList
               Dec → VarDec  |  VarDec ASSIGNOP Exp */
            Node *dec = declist->children[0];
            Node *vardec = dec->children[0];
            Node *idnode = tr_get_vardec_id(vardec);
            Type ftype = tr_vardec_to_type(vardec, base);
            FieldList f = new_field(idnode->sval, ftype, NULL);
            if (!head) head = tail = f; else { tail->tail = f; tail = f; }
            if (declist->nchild == 3) declist = declist->children[2];
            else declist = NULL;
        }
        deflist = deflist->children[1];  /* DefList → Def DefList */
    }
    return head;  /* 注意：返回 FieldList，但 type.kind 需为 STRUCTURE */
}

    /* 函数体内局部结构体的 DefList，复用上面逻辑 */
    FieldList head = NULL, tail = NULL;
    while (deflist && deflist->children[0]) {
        Node *def = deflist->children[0];
```

```c
            /* ParamDec → Specifier VarDec */

            gen_param(new_var(pid->sval));
            if (varlist->nchild == 3) varlist = varlist->children[2];
            else varlist = NULL;

    translate_compst(compst);

void translate_compst(Node *compst) {
    /* CompSt → LC DefList StmtList RC */
    Node *deflist = compst->children[1];
    Node *stmtlist = compst->children[2];

    /* DefList：局部变量。数组/结构体输出 DEC v_<name> <size> */
    /* type_size 在 translate_addr.c 实现（Task 7/9），这里先 extern */
    extern int type_size(Type t);
    while (deflist && deflist->children[0]) {
        Node *def = deflist->children[0];  /* Def → Specifier DecList SEMI */
        Node *spec = def->children[0];
        Node *declist = def->children[1];
        Type base = tr_specifier_to_type(spec);
        while (declist) {
            Node *dec = declist->children[0];  /* Dec → VarDec [ASSIGNOP Exp] */
            Node *vardec = dec->children[0];
            Node *idnode = tr_get_vardec_id(vardec);
            Type vtype = tr_vardec_to_type(vardec, base);
            translate_insert_var(idnode->sval, vtype);
            if (vtype->kind != BASIC) {
                gen_dec(new_var(idnode->sval), type_size(vtype));
            }
            if (declist->nchild == 3) declist = declist->children[2];
            else declist = NULL;
        }
        deflist = deflist->children[1];
    }

    /* StmtList：逐个翻译（Task 5 实现 translate_stmt，这里先占位循环） */
    while (stmtlist && stmtlist->children[0]) {
        translate_stmt(stmtlist->children[0]);
        stmtlist = stmtlist->children[1];
    }
}
```

```c
void translate_exp(Node *exp, Operand place) {
    /* Exp → INT */
    if (exp->children[0]->is_token && strcmp(exp->children[0]->name, "INT") == 0) {
        if (place.name || place.kind == OP_CONST) {  /* place 非 NULL */
            gen_assign(place, new_const((int)exp->children[0]->ival));
        }
        return;

    /* Exp → ID */
    if (is_id(exp)) {
        if (place.name) {
            VarEntry *e = translate_lookup_var(exp->children[0]->sval);
            gen_assign(place, new_var_from_ir(e ? e->ir_name : exp->children[0]->sval));
        }
        return;

    if (exp->nchild >= 2 && exp->children[1]->is_token
        && strcmp(exp->children[1]->name, "ASSIGNOP") == 0) {
        Node *lhs = exp->children[0];
        Node *rhs = exp->children[2];
        /* 仅处理左值为 ID（数组/结构体左值在 Task 7/9） */
        if (is_id(lhs)) {
            Operand t = new_temp();
            translate_exp(rhs, t);

            gen_assign(vlhs, t);

        return;

        if (strcmp(op, "PLUS") == 0) k = IR_ADD;

        return;
```

```c
        && strcmp(exp->children[1]->name, "RELOP") == 0) {
```

```c
int type_size(Type t) {
    if (!t) return 0;
```

```c
int cannot_translate = 0;

static int type_has_struct(Type t) {
    if (!t) return 0;
    switch (t->kind) {
        case BASIC: return 0;
        case ARRAY: return type_has_struct(t->u.array.elem);
        case STRUCTURE: return 1;
    }
    return 0;
}

   宏未开时，发现结构体 → cannot_translate=1。 */
void prepass_check(Node *root) {
#ifdef ENABLE_STRUCT
    return;  /* 开宏则不检测 */
#else
    Node *extdeflist = root->children[0];
    while (extdeflist && extdeflist->children[0]) {
        Node *extdef = extdeflist->children[0];
        Node *spec = extdef->children[0];
        Node *second = extdef->children[1];
        if (second && strcmp(second->name, "FunDec") == 0) {
            /* 形参 */
            if (second->nchild == 4) {

                    vl = (vl->nchild == 3) ? vl->children[2] : NULL;
                }
            }
            /* 局部变量 */
            Node *compst = extdef->children[2];
```

```c
        Type base_type; Operand base_addr = translate_addr_general(exp->children[0], &base_type);
        Type elem_type = base_type->u.array.elem;
        Operand t_idx = new_temp(); translate_exp(exp->children[2], t_idx);
        Operand t_off = new_temp(); gen_binop(IR_MUL, t_off, t_idx, new_const(type_size(elem_type)));
        Operand a = new_temp(); gen_binop(IR_ADD, a, base_addr, t_off);
        if (out_type) *out_type = elem_type;
        return a;
    }
    /* Exp DOT ID */
    if (exp->nchild == 3 && exp->children[1]->is_token && strcmp(exp->children[1]->name,"DOT")==0) {
        Type base_type; Operand base_addr = translate_addr_general(exp->children[0], &base_type);
        int off = 0;
        Type ftype = struct_field_offset(base_type->u.structure, exp->children[2]->sval, &off);
        Operand a = new_temp(); gen_binop(IR_ADD, a, base_addr, new_const(off));
        if (out_type) *out_type = ftype;
        return a;
    }
    Operand a = new_temp(); if (out_type) *out_type = NULL; return a;
}
```

```c
static int type_unsupported(Type t) {
    if (!t) return 0;
    switch (t->kind) {
        case BASIC: return 0;
        case ARRAY:
#ifdef ENABLE_HIGH_DIM_ARRAY
```

> ⚠️ 必做允许**一维数组变量**（A-3/B-1 都有 `int buf[6]`），但禁止**高维数组变量**和**任何数组作形参**。所以 prepass 要精确区分：

> - 变量是高维数组（≥2维）→ 必做不支持
> - 变量是一维数组 → 必做支持
>

```c
static int param_unsupported(Type t) {  /* 形参：数组/结构体均不支持 */
#ifndef ENABLE_STRUCT
    if (type_has_struct(t)) return 1;
#endif
#ifndef ENABLE_HIGH_DIM_ARRAY
    if (t && t->kind == ARRAY) return 1;  /* 数组作形参不支持 */
#endif
    return 0;
}
static int var_unsupported(Type t) {     /* 变量：高维数组/结构体不支持 */
#ifndef ENABLE_STRUCT
    if (type_has_struct(t)) return 1;
#endif
#ifndef ENABLE_HIGH_DIM_ARRAY
    if (t && t->kind == ARRAY && t->u.array.elem->kind == ARRAY) return 1;  /* ≥2维 */
#endif
    return 0;
}
```

- [ ] **Step 3: 数组参数 ARG &（E2-2/3）**
